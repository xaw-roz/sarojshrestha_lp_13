public class FindPrime {

    public static boolean isPrime(int n) {
        if (n == 2) return true;
        if (n % 2 == 0 || n == 1) return false;//even numbers are not prime except 2
        long limit = (long)Math.sqrt(n);
        for (int i = 3; i <= limit; i += 2)
            if (n % i == 0) return false;
        return true;
    }

    public static void main(String args[]) {
        int count = 1;
        int i = 3;
        while (count < 10001) {
            if (isPrime(i)) ++count;
            i += 2;//adding 2 to odd number gives odd number
        }
        System.out.println(i - 2);
    }
}
